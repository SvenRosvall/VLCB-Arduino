71template <typename E>
72void CircularBuffer<E>::put(const E &entry)
73{
74 buffer[head] = entry;
75
76 if (full)
77 {
78 // if the buffer is full, this put will overwrite the oldest item
79 tail = (tail + 1) % capacity;
80 ++numOverflows;
81 }
82
83 head = (head + 1) % capacity;
84 full = head == tail;
85 uint8_t size = bufUse();
86 if (size > hwm)
87 {
88 // Tracks high watermark (hwm)
89 hwm = size;
90 }
91 ++numPuts; // Counts how many events put to buffer.
92 // DEBUG_SERIAL << ">COE Puts = " << numPuts << " Size = " << size <<endl;
93}

96template <typename E>
97const E & CircularBuffer<E>::pop()
98{
99 uint8_t oldTail = tail;
100 full = false;
101 tail = (tail + 1) % capacity;
102 ++numGets; // Counts how many events got from buffer.
103 // DEBUG_SERIAL << ">COE Gets = " << numGets << endl;
104 return buffer[oldTail];
105}

132template <typename E>
133uint8_t CircularBuffer<E>::bufUse() const
134{
135 int8_t size = capacity;
136 if (!full)
137 {
138 size = head - tail;
139 if (size < 0)
140 {
141 size += capacity;
142 }
143 }
144 return size;
145}
